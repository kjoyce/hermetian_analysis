\problem{{\bf D'Angelo 1.40.} Assume that $f:\SS^1\to \CC$ is $k$ times continuously differentiable.  Show that there is a constant $C$ such that for $n>0$
$$
  |\hat f(n)| \le \frac C{n^k}.
$$
}

\begin{solution}
First, observe 
\begin{align*}
 \FF(f^{(j)})(n) &= \int_0^{2\pi} f^{(j)}(x) e^{-inx}\,dx \\
 &= \left. f^{(j-1)}(x)e^{-inx} \right|_{x=0}^{2\pi} + \frac {in}{2\pi} \int_0^{2\pi} f^{(j-1)}e^{-inx}\,dx\\
 &= 0 + in\FF(f^{(j-1)})(n).
\end{align*}
After $j-1$ more integration by parts, we have 
$$
  \FF(f^{(j)})(n) = (in)^j \FF(f)(n).
$$
Hence,
\begin{align*}
  \|\FF(f)\|_{\sup n} &= \frac{\|\FF(f^{(j)})\|_{\sup n}}{n^k} \le \frac{\|f^{(j)}\|_{L_1}}{n^k}
\end{align*}
which is finite since $f^{(j)}$ is continuous on the compact set $\SS^1$.
